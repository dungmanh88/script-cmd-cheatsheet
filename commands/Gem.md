# Path where gem-bame is installed
```
gem which <gem-name>
```

# Path where gem is installed
```
echo $GEM_HOME
or
gem environment gemdir
```

# Path used to search for gem
```
gem environment gempath
```

# Get gem version
```
gem --version
ruby --version
rails --version
bundle --version
rake --version
```

# Get installed gem
```
gem query --local
```
